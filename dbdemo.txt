CREATE TABLE renderjobs
( 
  `id` BIGINT(20) UNSIGNED NOT NULL AUTO_INCREMENT,
  `uuid` CHAR(16) NOT NULL,
  `fuid` CHAR(16) NOT NULL,
  `start` TIMESTAMP NOT NULL,
  `end` TIMESTAMP NOT NULL,
  `gputypecode` TINYINT(3) NOT NULL,
  `creation_date` DATE,
  `memo` VARCHAR(64),
  INDEX(`uuid`),
  INDEX(`fuid`),
  PRIMARY KEY (`id`)
)DEFAULT CHARACTER SET = utf8;;

CREATE TABLE user
( 
  `uuid` CHAR(16) NOT NULL,
  `passwd` CHAR(16),
  `mobile` VARCHAR(16),
  `group` CHAR(8),
  `level` TINYINT(2) UNSIGNED DEFAULT 1 NOT NULL ,
  `creation_date` DATE,
  `memo` VARCHAR(64),
  UNIQUE INDEX(`mobile`),
  PRIMARY KEY (`uuid`)
)DEFAULT CHARACTER SET = utf8;

CREATE TABLE task
( 
  `fuid` CHAR(16) NOT NULL,
  `uuid` CHAR(16) NOT NULL,
  `name` VARCHAR(64),
  `creation_ts` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `action` TINYINT(1) UNSIGNED DEFAULT 1 NOT NULL,
  `update_ts` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `memo` VARCHAR(128),
  INDEX(`uuid`),
  PRIMARY KEY (`fuid`)
)DEFAULT CHARACTER SET = utf8;
